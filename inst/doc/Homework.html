Rwms<-function(sigma,x0,N){
x<-matrix(NA,N,2);x[1,]<-c(x0,0);u<-runif(N)
for (i in 2:N) {
y <- rnorm(1, x[i-1,1], sigma)
c1=1/2*exp(-abs(y))
c2=1/2*exp(-abs(x[i-1,1]))
if (u[i] <= ( c1/c2 )){
x[i,1] <- y ;x[i,2]<-1}
else {
x[i,1] <- x[i-1,1];x[i,2]<-0
}
}
return(x)
}
